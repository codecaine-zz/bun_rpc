    async function escapeHTML() {
      const text = getInput();
      try {
        const result = await api.escapeHTML(text);
        setOutput(result);
        
        // Show a visual comparison
        const changes = [];
        if (text.includes('<')) changes.push('< ‚Üí <');
        if (text.includes('>')) changes.push('> ‚Üí >');
        if (text.includes('&')) changes.push('& ‚Üí &');
        if (text.includes('"')) changes.push('" ‚Üí "');
        if (text.includes("'")) changes.push("' ‚Üí '");
        
        const infoBox = document.getElementById('escapeInfo');
        const changesSpan = document.getElementById('escapeChanges');
        
        if (changes.length > 0) {
          changesSpan.textContent = changes.join(', ');
          infoBox.style.display = 'block';
        } else {
          changesSpan.textContent = 'No HTML characters found to escape';
          infoBox.style.display = 'block';
        }
      } catch (e) {
        alert(`Error: ${e.message}`);
      }
    }

    async function findReplace() {
      const text = getInput();
      const find = document.getElementById('findText').value;
      const replace = document.getElementById('replaceText').value;
      const caseSensitive = document.getElementById('caseSensitive').checked;

      if (!find) {
        alert('Please enter text to find');
        return;
      }

      try {
        const result = await api.findReplace(text, find, replace, caseSensitive);
        setOutput(result.result);
        alert(`Replaced ${result.count} occurrence(s)`);
      } catch (e) {
        alert(`Error: ${e.message}`);
      }
    }
